        id INTEGER PRIMARY KEY AUTOINCREMENT,

        last_scan_at INTEGER,
        is_active INTEGER DEFAULT 1,
        created_at INTEGER NOT NULL DEFAULT (unixepoch())
      );

      CREATE TABLE IF NOT EXISTS categories (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT UNIQUE NOT NULL,
        parent_id INTEGER REFERENCES categories(id),
        is_system INTEGER DEFAULT 0,
        sort_order INTEGER DEFAULT 0,
        created_at INTEGER NOT NULL DEFAULT (unixepoch())
      );

      CREATE TABLE IF NOT EXISTS samples (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        file_path TEXT UNIQUE NOT NULL,
        file_name TEXT NOT NULL,
        file_size INTEGER NOT NULL,
        file_hash TEXT NOT NULL,
        created_at INTEGER NOT NULL,
        modified_at INTEGER NOT NULL,
        duration REAL NOT NULL,
        sample_rate INTEGER NOT NULL,
        bit_rate INTEGER NOT NULL,
        channels INTEGER NOT NULL,
        bpm REAL,
        key TEXT,
        category_id INTEGER REFERENCES categories(id),
        waveform_data BLOB,
        is_favorite INTEGER DEFAULT 0,
        play_count INTEGER DEFAULT 0,
        last_played_at INTEGER,
        indexed_at INTEGER NOT NULL DEFAULT (unixepoch())
      );

      CREATE TABLE IF NOT EXISTS tags (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT UNIQUE NOT NULL,
        color TEXT DEFAULT '#1890ff',
        created_at INTEGER NOT NULL DEFAULT (unixepoch())
      );

      CREATE TABLE IF NOT EXISTS sample_tags (
        sample_id INTEGER NOT NULL REFERENCES samples(id),
        tag_id INTEGER NOT NULL REFERENCES tags(id),
        PRIMARY KEY (sample_id, tag_id)
      );

      CREATE TABLE IF NOT EXISTS classification_rules (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        pattern TEXT NOT NULL,
        rule_type TEXT NOT NULL,

### Task 6: IPC 通信框架

**目标：** 实现类型安全的主进程与渲染进程通信

**步骤：**

- [ ] **Step 1: 创建 main/ipcHandlers.ts**
  ```typescript
  import { ipcMain, IpcMainInvokeEvent } from 'electron';
  import { IPC_CHANNELS, IpcResponse } from '../shared/types/ipc.types';
  import { db } from './database';
  import { samples, categories, tags, watchedFolders } from '../drizzle/schema';
  import { eq, like, and, or } from 'drizzle-orm';

  type Handler<T = unknown, R = unknown> = (
    event: IpcMainInvokeEvent,
    data: T
  ) => Promise<R>;

  export function registerIpcHandlers(): void {
    // 获取采样列表
    ipcMain.handle(IPC_CHANNELS.GET_SAMPLES, async () => {
      try {
        const result = await db.select().from(samples).limit(100);
        return { success: true, data: result };
      } catch (error) {
        return { success: false, error: (error as Error).message };
      }
    });

    // 搜索采样
    ipcMain.handle(IPC_CHANNELS.SEARCH_SAMPLES, async (event, filters) => {
      try {
        // 实现搜索逻辑
        const result = await db.select().from(samples).limit(50);
        return { success: true, data: result };
      } catch (error) {
        return { success: false, error: (error as Error).message };
      }
    });

    // 获取分类
    ipcMain.handle(IPC_CHANNELS.GET_CATEGORIES, async () => {
      try {
        const result = await db.select().from(categories);
        return { success: true, data: result };
      } catch (error) {
        return { success: false, error: (error as Error).message };
      }
    });

    // 添加监控文件夹
    ipcMain.handle(IPC_CHANNELS.ADD_WATCHED_FOLDER, async (event, path: string) => {
      try {
        await db.insert(watchedFolders).values({ path });
        return { success: true };
      } catch (error) {
        return { success: false, error: (error as Error).message };
      }
    });

    // 其他 handlers...
  }
  ```

- [ ] **Step 2: 创建渲染进程 IPC 封装**
  文件：`renderer/src/utils/ipc.ts`
  ```typescript
  import { ipcRenderer } from 'electron';
  import { IPC_CHANNELS, IpcChannel, IpcResponse } from '../../../shared/types/ipc.types';

  export async function ipcInvoke<T = unknown, R = unknown>(
    channel: IpcChannel,
    data?: T
  ): Promise<R> {
    const response = await ipcRenderer.invoke(channel, data);
    return response;
  }

  export function ipcOn(channel: string, callback: (...args: unknown[]) => void): void {
    ipcRenderer.on(channel, callback);
  }

  export function ipcOff(channel: string, callback: (...args: unknown[]) => void): void {
    ipcRenderer.off(channel, callback);
  }
  ```

  import { IpcChannel } from '../shared/types/ipc.types';

  const electronAPI = {
    invoke: <T = unknown, R = unknown>(channel: IpcChannel, data?: T): Promise<R> => {
      return ipcRenderer.invoke(channel, data);
    },
    on: (channel: string, callback: (...args: unknown[]) => void) => {
      ipcRenderer.on(channel, callback);
    },
    off: (channel: string, callback: (...args: unknown[]) => void) => {
      ipcRenderer.off(channel, callback);
    },
  };

  contextBridge.exposeInMainWorld('electronAPI', electronAPI);

  export type ElectronAPI = typeof electronAPI;
