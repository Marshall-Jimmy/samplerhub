- [ ] **Step 2: 实现搜索函数**
  ```typescript
  import { db } from './database';
  import { sql } from 'drizzle-orm';
  import { SearchFilters } from '../shared/types/sample.types';

  interface SearchResult {
    items: Sample[];
    total: number;
    page: number;
    pageSize: number;
  }

  export async function searchSamples(
    filters: SearchFilters,
    page: number = 1,
    pageSize: number = 50
  ): Promise<SearchResult> {
    const conditions: string[] = [];
    const params: (string | number)[] = [];
    
    // 全文搜索
    if (filters.query) {
      conditions.push(`s.id IN (SELECT rowid FROM samples_fts WHERE samples_fts MATCH ?)`);
      params.push(filters.query);
    }
    
    // 分类筛选
    if (filters.categoryId) {
      conditions.push(`s.category_id = ?`);
      params.push(filters.categoryId);
    }
    
    // 时长范围
    if (filters.durationMin !== undefined) {
      conditions.push(`s.duration >= ?`);
      params.push(filters.durationMin);
    }
    if (filters.durationMax !== undefined) {
      conditions.push(`s.duration <= ?`);
      params.push(filters.durationMax);
    }
    
    // 采样率
    if (filters.sampleRate) {
      conditions.push(`s.sample_rate = ?`);
      params.push(filters.sampleRate);
    }
    
    // 比特率
    if (filters.bitRate) {
      conditions.push(`s.bit_rate = ?`);
      params.push(filters.bitRate);
    }
    
    // 声道数
    if (filters.channels) {
      conditions.push(`s.channels = ?`);
      params.push(filters.channels);
    }
    
    // BPM 范围
    if (filters.bpmMin !== undefined) {
      conditions.push(`s.bpm >= ?`);
      params.push(filters.bpmMin);
    }
    if (filters.bpmMax !== undefined) {
      conditions.push(`s.bpm <= ?`);
      params.push(filters.bpmMax);
    }
    
    // 调性
    if (filters.key) {
      conditions.push(`s.key = ?`);
      params.push(filters.key);
    }
    
    // 收藏
    if (filters.isFavorite !== undefined) {
      conditions.push(`s.is_favorite = ?`);
      params.push(filters.isFavorite ? 1 : 0);
    }
    
    // 标签筛选
    if (filters.tagIds && filters.tagIds.length > 0) {
      const placeholders = filters.tagIds.map(() => '?').join(',');
      conditions.push(`s.id IN (SELECT sample_id FROM sample_tags WHERE tag_id IN (${placeholders}))`);
      params.push(...filters.tagIds);
    }
    
    const whereClause = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : '';
    const offset = (page - 1) * pageSize;
    
    // 查询结果
    const results = await db.all(sql`
      SELECT s.*, c.name as category_name
      FROM samples s
      LEFT JOIN categories c ON s.category_id = c.id
      ${sql.raw(whereClause)}
      ORDER BY s.play_count DESC, s.file_name ASC
      LIMIT ${pageSize} OFFSET ${offset}
    `, ...params);
    
    // 查询总数
    const countResult = await db.get(sql`
      SELECT COUNT(*) as total FROM samples s
      ${sql.raw(whereClause)}
    `, ...params);
    
    return {
      items: results as Sample[],
      total: (countResult as { total: number })?.total || 0,
      page,
      pageSize,
    };
  }

  import { useState, useEffect } from 'react';

  export function useDebounce<T>(value: T, delay: number): T {
    const [debouncedValue, setDebouncedValue] = useState(value);

    useEffect(() => {
      const timer = setTimeout(() => {
        setDebouncedValue(value);
      }, delay);

      return () => {
        clearTimeout(timer);
      };
    }, [value, delay]);

    return debouncedValue;
  }
  ```

  interface DragData {

  export function setupDragAndDrop(
    element: HTMLElement, 
    sample: Sample,
    onDragStart?: () => void
  ): void {
    element.draggable = true;
    
    element.addEventListener('dragstart', (e: DragEvent) => {
      if (!e.dataTransfer) return;
      
      e.dataTransfer.effectAllowed = 'copy';
      
      // 设置文件路径（使用 file:// 协议）
      const fileUrl = `file://${sample.filePath}`;
      
      // 设置多种数据格式以提高兼容性
      e.dataTransfer.setData('text/uri-list', fileUrl);
      e.dataTransfer.setData('text/plain', sample.filePath);
      e.dataTransfer.setData('application/x-file-url', fileUrl);
      
      // 设置拖拽图像
      const dragImage = createDragImage(sample);
      e.dataTransfer.setDragImage(dragImage, 0, 0);
      
      onDragStart?.();
    });
  }

  export function setupMultiDrag(
    container: HTMLElement,
    getSelectedSamples: () => Sample[]
  ): void {
    container.addEventListener('dragstart', (e: DragEvent) => {
      if (!e.dataTransfer) return;
      
      const selectedSamples = getSelectedSamples();
      
      if (selectedSamples.length === 1) {
        const sample = selectedSamples[0];
        const fileUrl = `file://${sample.filePath}`;
        e.dataTransfer.setData('text/uri-list', fileUrl);
        e.dataTransfer.setData('text/plain', sample.filePath);
      } else {
        // 多个文件：使用换行分隔的路径列表
        const paths = selectedSamples.map(s => s.filePath).join('\n');
        const urls = selectedSamples.map(s => `file://${s.filePath}`).join('\n');
        e.dataTransfer.setData('text/plain', paths);
        e.dataTransfer.setData('text/uri-list', urls);

    });

  import { db } from './database';
  import { samples } from '../drizzle/schema';
  import { eq } from 'drizzle-orm';

  export async function toggleFavorite(sampleId: number): Promise<boolean> {
    const sample = await db.select().from(samples).where(eq(samples.id, sampleId)).get();
    
    if (!sample) {
      throw new Error('Sample not found');
    }

    const newFavorite = !sample.isFavorite;
    
    await db.update(samples)
      .set({ isFavorite: newFavorite })
      .where(eq(samples.id, sampleId));
    
    return newFavorite;
  }

  export async function getFavorites(): Promise<Sample[]> {
    return db.select().from(samples).where(eq(samples.isFavorite, true));
  }

  // main/recent.ts
  import { db } from './database';
  import { recentSamples, samples } from '../drizzle/schema';
  import { eq, desc, sql } from 'drizzle-orm';

  const MAX_RECENT = 50;

  export async function addRecent(sampleId: number): Promise<void> {
    // 添加新记录
    await db.insert(recentSamples).values({ sampleId });
    
    // 清理旧记录，只保留最近 50 条
    const count = await db.select({ count: sql<number>`count(*)` }).from(recentSamples);
    
    if (count[0].count > MAX_RECENT) {
      const toDelete = await db.select()
        .from(recentSamples)
        .orderBy(desc(recentSamples.playedAt))
        .offset(MAX_RECENT);
      
      for (const record of toDelete) {
        await db.delete(recentSamples).where(eq(recentSamples.id, record.id));
      }
    }
  }

  export async function getRecent(limit: number = 20): Promise<Sample[]> {
    const result = await db.all(sql`
      SELECT s.* FROM samples s
      JOIN recent_samples r ON s.id = r.sample_id
      ORDER BY r.played_at DESC
      LIMIT ${limit}
    `);
    
    return result as Sample[];
  }

              style={{ borderStyle: 'dashed', cursor: 'pointer' }}
              onClick={() => setIsAdding(true)}
            >
              <PlusOutlined /> 新建标签
            </Tag>
          )}
        </Space>
      </div>
    );
  };

  export async function importLibrary(): Promise<boolean> {

      ],
      properties: ['openFile'],
    });

    if (!filePaths || filePaths.length === 0) return false;

    const filePath = filePaths[0];
    const content = await readFile(filePath, 'utf-8');
    const backupData: BackupData = JSON.parse(content);

    // 验证版本
    if (!backupData.version) {
      throw new Error('Invalid backup file');
    }

    // 清空现有数据（可选：提供合并选项）
    // await db.delete(sampleTags);
    // await db.delete(samples);
    // ...

    // 导入数据
    for (const category of backupData.categories) {
      await db.insert(categories).values(category).onConflictDoNothing();
    }

    for (const tag of backupData.tags) {
      await db.insert(tags).values(tag).onConflictDoNothing();
    }

    for (const sample of backupData.samples) {
      await db.insert(samples).values(sample).onConflictDoNothing();
    }
