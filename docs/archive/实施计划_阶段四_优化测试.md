  ```typescript
  // main/fileScanner.ts - 优化版
  
  import { Worker } from 'worker_threads';
  import os from 'os';

  const CPU_COUNT = os.cpus().length;
  const BATCH_SIZE = 100;

  export async function scanFolderOptimized(
    folderPath: string,
    onProgress?: (progress: ScanProgress) => void
  ): Promise<void> {
    // 1. 快速收集文件列表（不计算哈希）
    const filePaths = await fastCollectFiles(folderPath);
    
    // 2. 分批处理，使用 Worker 线程池
    const workerPool = new Array(CPU_COUNT).fill(null).map(() => 
      new Worker('./worker/scanBatch.worker.js')
    );
    
    let completedBatches = 0;
    const totalBatches = Math.ceil(filePaths.length / BATCH_SIZE);
    
    for (let i = 0; i < filePaths.length; i += BATCH_SIZE) {
      const batch = filePaths.slice(i, i + BATCH_SIZE);
      const workerIndex = (i / BATCH_SIZE) % CPU_COUNT;
      
      workerPool[workerIndex].postMessage({
        batch,
        batchIndex: i / BATCH_SIZE,
      });
      
      workerPool[workerIndex].once('message', (result) => {
        completedBatches++;
        onProgress?.({
          current: completedBatches * BATCH_SIZE,
          total: filePaths.length,
          currentFile: result.lastFile,
          phase: 'parsing',
        });
      });
    }
    
    // 等待所有 Worker 完成
    await Promise.all(workerPool.map(worker => 
      new Promise<void>((resolve) => {
        worker.on('exit', resolve);
      })
    ));
  }

  async function fastCollectFiles(dir: string): Promise<string[]> {
    const files: string[] = [];
    
    async function traverse(currentDir: string): Promise<void> {
      const entries = await readdir(currentDir, { withFileTypes: true });
      
      // 使用 Promise.all 并行处理
      await Promise.all(entries.map(async (entry) => {
        const fullPath = join(currentDir, entry.name);
        
        if (entry.isDirectory()) {
          await traverse(fullPath);
        } else if (entry.isFile() && AUDIO_EXTENSIONS.has(extname(entry.name).toLowerCase())) {
          files.push(fullPath);
        }
      }));
    }
    
    await traverse(dir);
    return files;
  }
  ```

- [ ] **Step 2: 搜索优化**
  ```typescript
  // main/search.ts - 添加索引和缓存
  
  import NodeCache from 'node-cache';

  const searchCache = new NodeCache({ stdTTL: 300 }); // 5分钟缓存

  export async function searchSamplesCached(
    filters: SearchFilters,
    page: number = 1,
    pageSize: number = 50
  ): Promise<SearchResult> {
    const cacheKey = JSON.stringify({ filters, page, pageSize });
    
    // 检查缓存
    const cached = searchCache.get<SearchResult>(cacheKey);
    if (cached) {
      return cached;
    }
    
    // 执行搜索
    const result = await searchSamples(filters, page, pageSize);
    
    // 存入缓存
    searchCache.set(cacheKey, result);
    
    return result;
  }

  // 清除缓存（在数据变更时调用）
  export function invalidateSearchCache(): void {
    searchCache.flushAll();
  }
  ```

  // renderer/src/components/ScanProgress.tsx
  import React from 'react';
  import { Progress, Typography, Space } from 'antd';
  import { ScanProgress as ScanProgressType } from '../../../shared/types/sample.types';

  interface ScanProgressProps {
    progress: ScanProgressType | null;
  }

  const PHASE_LABELS: Record<string, string> = {
    scanning: '扫描文件中...',
    parsing: '解析音频元数据...',
    classifying: '智能分类中...',
    complete: '扫描完成',
  };

  const ScanProgress: React.FC<ScanProgressProps> = ({ progress }) => {
    if (!progress) return null;

    const percent = progress.total > 0 
      ? Math.round((progress.current / progress.total) * 100) 
      : 0;

    return (
      <div style={{ padding: 16, backgroundColor: '#f6ffed', borderRadius: 8 }}>
        <Space direction="vertical" style={{ width: '100%' }}>
          <Typography.Text strong>
            {PHASE_LABELS[progress.phase] || '处理中...'}
          </Typography.Text>
          <Progress percent={percent} status={progress.phase === 'complete' ? 'success' : 'active'} />
          <Typography.Text type="secondary">
            {progress.current} / {progress.total} - {progress.currentFile}
          </Typography.Text>
        </Space>
      </div>
    );
  };

  export default ScanProgress;

  // renderer/src/hooks/useKeyboardShortcuts.ts
  import { useEffect } from 'react';

  interface ShortcutMap {
    [key: string]: () => void;
  }

  export function useKeyboardShortcuts(shortcuts: ShortcutMap): void {
    useEffect(() => {
      const handleKeyDown = (e: KeyboardEvent) => {
        const key = `${e.ctrlKey || e.metaKey ? 'Ctrl+' : ''}${e.shiftKey ? 'Shift+' : ''}${e.key}`;
        
        if (shortcuts[key]) {
          e.preventDefault();
          shortcuts[key]();
        }
      };

      window.addEventListener('keydown', handleKeyDown);
      return () => window.removeEventListener('keydown', handleKeyDown);
    }, [shortcuts]);
  }

### Task 5: Bug 修复清单

**常见问题和解决方案：**

| 问题 | 解决方案 |
|------|----------|
| 中文路径乱码 | 使用 `path.normalize()` 和 `decodeURIComponent()` |
| 大文件解析内存溢出 | 使用流式解析，限制 Worker 内存 |

| 音频播放卡顿 | 使用 Howler.js 的 `html5: true` 模式 |
| 波形显示空白 | 检查 wavesurfer.js 的 CORS 配置 |
