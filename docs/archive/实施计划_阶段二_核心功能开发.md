  ```typescript
  import { readdir, stat } from 'fs/promises';
  import { join, extname } from 'path';
  import { AUDIO_EXTENSIONS } from '../shared/constants/audioFormats';

  interface FileInfo {
    path: string;
    name: string;
    size: number;
    modifiedAt: Date;
    hash: string;
  }

  export async function collectAudioFiles(dir: string): Promise<FileInfo[]> {
    const files: FileInfo[] = [];
    
    async function traverse(currentDir: string): Promise<void> {
      const entries = await readdir(currentDir, { withFileTypes: true });
      
      for (const entry of entries) {
        const fullPath = join(currentDir, entry.name);
        
        if (entry.isDirectory()) {
          await traverse(fullPath);
        } else if (entry.isFile() && AUDIO_EXTENSIONS.has(extname(entry.name).toLowerCase())) {
          const stats = await stat(fullPath);
          const hash = await computeFileHash(fullPath, stats);
          
          files.push({
            path: fullPath,
            name: entry.name,
            size: stats.size,
            modifiedAt: stats.mtime,
            hash,
          });
        }
      }
    }
    
    await traverse(dir);
    return files;
  }

  import { createHash } from 'crypto';

  async function computeFileHash(filePath: string, stats: Stats): Promise<string> {
    const hash = createHash('md5');
    hash.update(`${filePath}:${stats.size}:${stats.mtime.getTime()}`);
    return hash.digest('hex');
  }
  ```

- [ ] **Step 3: 实现增量扫描逻辑**
  ```typescript
  import { db } from './database';
  import { samples, watchedFolders } from '../drizzle/schema';
  import { eq } from 'drizzle-orm';
  import { ScanProgress } from '../shared/types/sample.types';

  export async function scanFolder(
    folderPath: string, 
    onProgress?: (progress: ScanProgress) => void
  ): Promise<{ added: number; updated: number; deleted: number }> {
    // 1. 收集所有音频文件
    onProgress?.({ current: 0, total: 0, currentFile: '', phase: 'scanning' });
    const files = await collectAudioFiles(folderPath);
    
    // 2. 获取数据库中已有记录
    const existingSamples = await db.select().from(samples);
    const existingMap = new Map(existingSamples.map(s => [s.filePath, s]));
    
    // 3. 计算差异
    const toAdd: FileInfo[] = [];
    const toUpdate: FileInfo[] = [];
    const toDelete: number[] = [];
    
    for (const file of files) {
      const existing = existingMap.get(file.path);
      if (!existing) {
        toAdd.push(file);
      } else if (existing.fileHash !== file.hash) {
        toUpdate.push(file);
      }
      existingMap.delete(file.path);
    }
    
    // 剩余的就是已删除的文件
    for (const [, sample] of existingMap) {
      toDelete.push(sample.id);
    }
    
    // 4. 批量处理
    await processBatch(toAdd, toUpdate, toDelete, onProgress);
    
    // 5. 更新扫描时间
    await db.update(watchedFolders)
      .set({ lastScanAt: new Date() })
      .where(eq(watchedFolders.path, folderPath));
    
    return {
      added: toAdd.length,
      updated: toUpdate.length,
      deleted: toDelete.length,
    };
  }
  ```

- [ ] **Step 4: 实现批量数据库操作**
  ```typescript
  async function processBatch(
    toAdd: FileInfo[],
    toUpdate: FileInfo[],
    toDelete: number[],
    onProgress?: (progress: ScanProgress) => void
  ): Promise<void> {
    const total = toAdd.length + toUpdate.length + toDelete.length;
    let current = 0;
    
    // 批量插入
    if (toAdd.length > 0) {
      onProgress?.({ 
        current, 
        total, 
        currentFile: toAdd[0]?.name || '', 
        phase: 'parsing' 
      });
      
      // 使用事务批量插入
      const insertStmt = db.insert(samples).prepare();
      
      for (const file of toAdd) {
        await insertStmt.execute({
          filePath: file.path,
          fileName: file.name,
          fileSize: file.size,
          fileHash: file.hash,
          createdAt: new Date(),
          modifiedAt: file.modifiedAt,
          duration: 0, // 待解析
          sampleRate: 0,
          bitRate: 0,
          channels: 0,
        });
        
        current++;
        onProgress?.({ 
          current, 
          total, 
          currentFile: file.name, 
          phase: 'parsing' 
        });
      }
    }
    
    // 批量更新
    for (const file of toUpdate) {
      await db.update(samples)
        .set({
          fileSize: file.size,
          fileHash: file.hash,
          modifiedAt: file.modifiedAt,
        })
        .where(eq(samples.filePath, file.path));
      
      current++;
      onProgress?.({ current, total, currentFile: file.name, phase: 'parsing' });
    }
    
    // 批量删除
    if (toDelete.length > 0) {
      await db.delete(samples).where(inArray(samples.id, toDelete));
      current += toDelete.length;
    }
    
    onProgress?.({ current, total, currentFile: '', phase: 'complete' });
  }
  ```

  ```typescript
  import { parentPort } from 'worker_threads';
  import { parseFile } from 'music-metadata';

  interface ParseTask {
    filePath: string;
    sampleId: number;
  }

  interface ParseResult {
    sampleId: number;
    duration: number;
    sampleRate: number;
    bitRate: number;
    channels: number;
    bpm: number | null;
    key: string | null;
    waveformData: number[] | null;
    error?: string;
  }

  parentPort?.on('message', async (task: ParseTask) => {
    try {
      const metadata = await parseFile(task.filePath, {
        duration: true,
        skipPostHeaders: true,
      });
      
      const format = metadata.format;
      const common = metadata.common;
      
      // 生成简化波形数据（100个点）
      const waveformData = await generateWaveform(task.filePath);
      
      const result: ParseResult = {
        sampleId: task.sampleId,
        duration: format.duration || 0,
        sampleRate: format.sampleRate || 0,
        bitRate: format.bitrate || 0,
        channels: format.numberOfChannels || 0,
        bpm: common.bpm || null,
        key: common.key || null,
        waveformData,
      };
      
      parentPort?.postMessage(result);
    } catch (error) {
      parentPort?.postMessage({
        sampleId: task.sampleId,
        error: error instanceof Error ? error.message : 'Unknown error',
      });
    }
  });

  async function generateWaveform(filePath: string): Promise<number[]> {
    // 简化实现：返回100个随机数据点
    // 实际实现可以使用 audio-decode 库解析音频数据
    const POINTS = 100;
    return Array.from({ length: POINTS }, () => Math.random());
  }
  ```

- [ ] **Step 2: 创建 Worker 池管理器**
  ```typescript
  // main/workerPool.ts
  import { Worker } from 'worker_threads';
  import { db } from './database';
  import { samples } from '../drizzle/schema';
  import { eq } from 'drizzle-orm';

  const WORKER_COUNT = 4;

  class WorkerPool {
    private workers: Worker[] = [];
    private queue: Array<{ task: unknown; resolve: (value: unknown) => void }> = [];
    private busyWorkers = new Set<number>();

    constructor() {
      for (let i = 0; i < WORKER_COUNT; i++) {
        const worker = new Worker('./worker/audioParse.worker.js');
        worker.on('message', (result) => this.handleResult(i, result));
        this.workers.push(worker);
      }
    }

    async execute(task: unknown): Promise<unknown> {
      return new Promise((resolve) => {
        this.queue.push({ task, resolve });
        this.processQueue();
      });
    }

    private processQueue(): void {
      for (let i = 0; i < this.workers.length; i++) {
        if (!this.busyWorkers.has(i) && this.queue.length > 0) {
          const { task, resolve } = this.queue.shift()!;
          this.busyWorkers.add(i);
          this.workers[i].postMessage(task);
        }
      }
    }

    private handleResult(workerIndex: number, result: unknown): void {
      this.busyWorkers.delete(workerIndex);
      // 处理结果并更新数据库
      this.processQueue();
    }

    terminate(): void {
      this.workers.forEach(worker => worker.terminate());
    }
  }

  export const workerPool = new WorkerPool();
  ```

  ```typescript
  interface ClassificationResult {
    categoryId: number | null;
    matchedRules: ClassificationRule[];
  }

  export function classifySample(sample: Sample, rules: ClassificationRule[]): ClassificationResult {
    const matches: ClassificationRule[] = [];
    
    for (const rule of rules) {
      if (!rule.isActive) continue;
      
      const isMatch = matchRule(sample, rule);
      if (isMatch) {
        matches.push(rule);
      }
    }
    
    if (matches.length === 0) {
      return { categoryId: null, matchedRules: [] };
    }
    
    // 按优先级排序，取最高优先级的规则
    matches.sort((a, b) => b.priority - a.priority);
    
    return {
      categoryId: matches[0].targetCategoryId,
      matchedRules: matches,
    };
  }

  function matchRule(sample: Sample, rule: ClassificationRule): boolean {
    const searchText = `${sample.fileName} ${sample.filePath}`.toLowerCase();
    
    switch (rule.ruleType) {
      case 'regex':
        try {
          const regex = new RegExp(rule.pattern, 'i');
          return regex.test(searchText);
        } catch {
          return false;
        }
      
      case 'keyword':
        const keywords = rule.pattern.split('|').map(k => k.trim().toLowerCase());
        return keywords.some(keyword => searchText.includes(keyword));
      
      case 'folder':
        const folderPattern = rule.pattern.toLowerCase();
        return sample.filePath.toLowerCase().includes(folderPattern);
      
      default:
        return false;
    }
  }
  ```

  ```typescript
        <Space direction="vertical" style={{ width: '100%' }}>
          <Slider
            value={progress}
            onChange={handleSeek}
            tooltip={{ formatter: (value) => formatTime((value / 100) * duration) }}
          />
          
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
            <Typography.Text type="secondary">
              {formatTime(currentTime)} / {formatTime(duration)}
            </Typography.Text>
            
            <Space>
              <Button icon={<StepBackwardOutlined />} />
              <Button
                type="primary"
                shape="circle"
                icon={isPlaying ? <PauseCircleOutlined /> : <PlayCircleOutlined />}
                onClick={togglePlay}
                size="large"
              />
              <Button icon={<StepForwardOutlined />} />
            </Space>
          </div>
        </Space>
      </div>
    );
  };
  ```

### Task 7: 波形可视化（wavesurfer.js）

**目标：** 集成 wavesurfer.js 显示音频波形

**文件：** `renderer/src/components/WaveformViewer.tsx`

**步骤：**

- [ ] **Step 1: 创建波形可视化组件**
  ```typescript
  import React, { useEffect, useRef, useState } from 'react';
  import WaveSurfer from 'wavesurfer.js';
  import { Button, Space } from 'antd';
  import { PlayCircleOutlined, PauseCircleOutlined } from '@ant-design/icons';
  import { Sample } from '../../../shared/types/sample.types';

  interface WaveformViewerProps {
    sample: Sample | null;
    height?: number;
  }

  const WaveformViewer: React.FC<WaveformViewerProps> = ({ 
    sample, 
    height = 128 
  }) => {
    const containerRef = useRef<HTMLDivElement>(null);
    const waveSurferRef = useRef<WaveSurfer | null>(null);
    const [isPlaying, setIsPlaying] = useState(false);

    useEffect(() => {
      if (!sample || !containerRef.current) return;

      // 创建 WaveSurfer 实例
      const waveSurfer = WaveSurfer.create({
        container: containerRef.current,
        waveColor: '#1890ff',
        progressColor: '#096dd9',
        cursorColor: '#ff4d4f',
        barWidth: 2,
        barGap: 1,
        barRadius: 2,
        height,
        normalize: true,
      });

      waveSurfer.load(sample.filePath);

      waveSurfer.on('ready', () => {
        console.log('Waveform ready');
      });

      waveSurfer.on('play', () => setIsPlaying(true));
      waveSurfer.on('pause', () => setIsPlaying(false));
      waveSurfer.on('finish', () => setIsPlaying(false));

      waveSurferRef.current = waveSurfer;

      return () => {
        waveSurfer.destroy();
      };
    }, [sample, height]);

    const togglePlay = () => {
      waveSurferRef.current?.playPause();
    };

    if (!sample) {
      return (
        <div 
          style={{ 
            height, 
            backgroundColor: '#f0f2f5',
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            borderRadius: 4,
          }}
        >
          选择采样查看波形
        </div>
      );
    }

    return (
      <div>
        <div ref={containerRef} style={{ width: '100%' }} />
        <Space style={{ marginTop: 8 }}>
          <Button
            type="primary"
            icon={isPlaying ? <PauseCircleOutlined /> : <PlayCircleOutlined />}
            onClick={togglePlay}
          >
            {isPlaying ? '暂停' : '播放'}
          </Button>
        </Space>
      </div>
    );
  };

  export default WaveformViewer;
  ```

### Task 8: 采样库页面

**目标：** 整合所有组件，构建完整的采样库页面

**文件：** `renderer/src/pages/LibraryPage.tsx`

**步骤：**

- [ ] **Step 1: 创建页面组件**
  ```typescript
  import React, { useState, useCallback } from 'react';
  import { Row, Col, Card, Input, Tree, Button, Space } from 'antd';
  import { SearchOutlined, FolderAddOutlined } from '@ant-design/icons';
  import { useQuery } from '@tanstack/react-query';
  import { Sample, Category } from '../../../shared/types/sample.types';
  import SampleList from '../components/SampleList';
  import AudioPlayer from '../components/AudioPlayer';
  import WaveformViewer from '../components/WaveformViewer';
  import { ipcInvoke } from '../utils/ipc';
  import { IPC_CHANNELS } from '../../../shared/types/ipc.types';

  const LibraryPage: React.FC = () => {
    const [selectedCategory, setSelectedCategory] = useState<number | null>(null);
    const [searchQuery, setSearchQuery] = useState('');
    const [selectedSamples, setSelectedSamples] = useState<Set<number>>(new Set());
    const [playingSample, setPlayingSample] = useState<Sample | null>(null);
    const [waveformSample, setWaveformSample] = useState<Sample | null>(null);

    // 获取分类列表
    const { data: categories } = useQuery({
      queryKey: ['categories'],
      queryFn: async () => {
        const response = await ipcInvoke(IPC_CHANNELS.GET_CATEGORIES);
        return response.data as Category[];
      },
    });

    // 获取采样列表
    const { data: samples } = useQuery({
      queryKey: ['samples', selectedCategory, searchQuery],
      queryFn: async () => {
        const response = await ipcInvoke(IPC_CHANNELS.SEARCH_SAMPLES, {
          categoryId: selectedCategory,
          query: searchQuery,
        });
        return response.data as Sample[];
      },
    });

    const handleSelect = useCallback((id: number, multi: boolean) => {
      setSelectedSamples(prev => {
        const newSet = new Set(prev);
        if (multi) {
          if (newSet.has(id)) {
            newSet.delete(id);
          } else {
            newSet.add(id);
          }
        } else {
          newSet.clear();
          newSet.add(id);
        }
        return newSet;
      });
    }, []);

    const handlePlay = useCallback((sample: Sample) => {
      setPlayingSample(sample);
      setWaveformSample(sample);
    }, []);

    const handleDragStart = useCallback((sample: Sample) => {
      // 拖拽逻辑已在 SampleCard 中处理
      console.log('Drag start:', sample.fileName);
    }, []);

    const treeData = categories?.map(cat => ({
      key: cat.id,
      title: cat.name,
      children: cat.children?.map(child => ({
        key: child.id,
        title: child.name,
      })),
    }));

    return (
      <Row gutter={16} style={{ height: '100%' }}>
        {/* 左侧分类树 */}
        <Col span={5}>
          <Card 
            title="分类" 
            extra={
              <Button type="link" icon={<FolderAddOutlined />}>
                添加文件夹
              </Button>
            }
          >
            <Tree
              treeData={treeData}
              onSelect={(keys) => setSelectedCategory(keys[0] as number)}
            />
          </Card>
        </Col>

        {/* 中间采样列表 */}
        <Col span={11}>
          <Card>
            <Input
              prefix={<SearchOutlined />}
              placeholder="搜索采样..."
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              style={{ marginBottom: 16 }}
            />
            
            <div style={{ height: 'calc(100vh - 300px)' }}>
              <SampleList
                samples={samples || []}
                selectedIds={selectedSamples}
                onSelect={handleSelect}
                onPlay={handlePlay}
                onDragStart={handleDragStart}
              />
            </div>
          </Card>
        </Col>

        {/* 右侧详情面板 */}
        <Col span={8}>
          <Space direction="vertical" style={{ width: '100%' }} size="large">
            <Card title="波形预览">
              <WaveformViewer sample={waveformSample} />
            </Card>
            
            <Card title="播放器">
              <AudioPlayer sample={playingSample} />
            </Card>
          </Space>
        </Col>
      </Row>
    );
  };

  export default LibraryPage;
  ```
